	void OnFrame(float timeSlice)
	{	
		if (IsMMSDisplayedOnAnyPage()) flirUtil.UpdateCamera(timeSlice);
		if(!m_VehicleSignals) return;
		if (!camera) return;
		

		UpdateCamera();
		if(cameraStabilized)
		{
			StabilizeMMS();
		}

		if(enableDebug)
		{
			DiagMaterialUtil.update();
		}
	}
	
	override protected void EOnPostFrame(IEntity owner, float timeSlice)
    {
		OnFrame(timeSlice);
    }

	void OnReceiveUpdateCameraData(float vertical, float horizontal, bool stabilized, vector stabilizeTargetPos, int zoomLevel, vector mmsStartReceived, vector mmsDirReceived)
	{
		if(mmsControlKeyPressed)
		{
			return;
		}
		cameraVerticalOrientation = GetLimitAdjustedVerticalOrientation(vertical);
		cameraHorizontalOrientation = horizontal;
		cameraStabilized = stabilized;
		targetPosition = stabilizeTargetPos;
		m_VehicleSignals.SetSignalValue(mmsXSignal, cameraHorizontalOrientation);
		m_VehicleSignals.SetSignalValue(mmsYSignal, cameraVerticalOrientation);
		mmsStart = mmsStartReceived;
		mmsCameraDir = mmsDirReceived;
		
		currentZoomLevel = zoomLevel;
		UpdateCameraZoom();
		UpdateCamera();
	}

	protected void PilotCameraUp()
	{
		if (!camera) return;
		
		cameraStabilizedOnEntity = false;
		mmsControlKeyPressed = true;
		if(cameraStabilized)
		{
			targetLocked = false;
			cameraStabilized = false;
			cameraStabilizedEntity = null;
			UpdateRotationY(true);
			UpdateCamera();
			GetCameraTargetPos();
			cameraStabilized = true;
			SendRemoteUpdate();
		} else
		{
			UpdateRotationY(true);
			UpdateCamera();
			SendRemoteUpdate();
		}
	}

	private void UpdateRotationX(bool rotateLeft)
	{
		
		cameraRotStepValue = Math.Lerp(cameraRotStepValue, cameraRotStep, cameraAcceleration);
		if (rotateLeft)
			cameraHorizontalOrientation += cameraRotStepValue;
		else 
			cameraHorizontalOrientation -= Math.Lerp(cameraRotStepValue, cameraRotStepValue, cameraAcceleration);
		m_VehicleSignals.SetSignalValue(mmsXSignal, cameraHorizontalOrientation);
	}
	
	private void UpdateRotationY(bool rotateUp)
	{
		cameraRotStepValue = Math.Lerp(cameraRotStepValue, cameraRotStep, cameraAcceleration);
		if (rotateUp)
			cameraVerticalOrientation += cameraRotStepValue;
		else 
			cameraVerticalOrientation -= cameraRotStepValue;
		
		cameraVerticalOrientation = GetLimitAdjustedVerticalOrientation(cameraVerticalOrientation);
		m_VehicleSignals.SetSignalValue(mmsYSignal, cameraVerticalOrientation);
	}

	protected void PilotCameraZoomCycle()
	{
		if (!camera) return;
		currentZoomLevel += 1;

		UpdateCameraZoom();
		SendRemoteUpdate();
	}

	protected void PilotCameraLock()
	{
		if (!camera) return;
		cameraStabilized = !cameraStabilized;
		if(cameraStabilized)
		{
			GetCameraTargetPos(true);
		} else
		{
			targetLocked = false;
			cameraStabilizedOnEntity = false;
			distanceToTarget = -1;
		}
		SendRemoteUpdate();
	}

	protected void PilotCameraForward()
	{
		if (!camera) return;

		cameraStabilized = false;
		targetLocked = false;
		cameraStabilizedOnEntity = false;
		targetPosition = {};
		distanceToTarget = -1;

		cameraHorizontalOrientation = 0;
		cameraVerticalOrientation = 0;
		m_VehicleSignals.SetSignalValue(mmsXSignal, cameraHorizontalOrientation);
		m_VehicleSignals.SetSignalValue(mmsYSignal, cameraVerticalOrientation);
		SendRemoteUpdate();
	}

	protected void UpdateCameraZoom()
	{
		if (!IsVehicleCrew())
			return;
		
		float fov;
		if (currentZoomLevel > 5)
		{
			currentZoomLevel = 1;
		}
		switch (currentZoomLevel)
		{
			case 1:
			{
				fov = cameraFOV1;
				cameraRotStep = cameraFOV1MoveRate;
				break;
			}
			case 2:
			{
				fov = cameraFOV2;
				cameraRotStep = cameraFOV2MoveRate;
				break;
			}
			case 3:
			{
				fov = cameraFOV3;
				cameraRotStep = cameraFOV3MoveRate;
				break;
			}
			case 4:
			{
				fov = cameraFOV4;
				cameraRotStep = cameraFOV4MoveRate;
				break;
			}
			case 5:
			{
				fov = cameraFOV5;
				cameraRotStep = cameraFOV5MoveRate;
				break;
			}
		}

		GetGame().GetWorld().SetCameraVerticalFOV(cameraIndex, fov);
	}

	private void GetCameraTargetPos(bool tryPointLock = false)
	{
		if(targetLocked || !camera)
		{
			return;
		}
		// Initial update of vectors (before per-frame)
		UpdateMMSVectors();
		vector mmsDirection = camera.GetYawPitchRoll().AnglesToVector();
		RaycastResult traceResult = RayCastFromCamera(mmsDirection);

		targetPosition = mmsStart + (mmsDirection * 15) + (mmsDirection * maxRaycastDistanceForCameraStabilization * traceResult.distance);
		targetLocked = true;

		//Shape targetSphere = Shape.CreateSphere(Color.RED, ShapeFlags.DEFAULT, targetPosition, 3);
		//m_aDebugShapes.Insert(targetSphere);
		//m_aDebugShapes.Remove(0);
	}

	protected void UpdateCamera()
	{
		if (!camera) return;
		
		if (!IsVehicleCrew())
			return;
		
		camera.SetYawPitchRoll(Vector(-cameraHorizontalOrientation, cameraVerticalOrientation, 0));
		GetGame().GetWorld().SetCameraFarPlane(cameraIndex, GetGame().GetViewDistance());
	}

	protected void StabilizeMMS()
	{
		UpdateMMSVectors();
		RotateCamToTarget(targetPosition);
	}

	private void RotateCamToTarget(vector targetPos)
	{
		if (targetPos == {})
			return;
		
		
		vector transform[4];
		camera.GetTransform(transform);

		vector rotationMatrix[4];
		SCR_Math3D.LookAt(mmsStart, targetPos, vector.Up, rotationMatrix);

		float quat[4];
		Math3D.MatrixToQuat(rotationMatrix, quat);

		vector currentAngles = camera.GetYawPitchRoll();
		vector angles = Math3D.QuatToAngles(quat);

		Math3D.AnglesToMatrix(angles, transform);
		camera.SetTransform(transform);

		vector mmsDirection = camera.GetYawPitchRoll().AnglesToVector();
		vector mmsDiff = camera.GetLocalYawPitchRoll();

		// Rotate model
		cameraHorizontalOrientation = -mmsDiff[0];
		cameraVerticalOrientation = GetLimitAdjustedVerticalOrientation(mmsDiff[1]);
		m_VehicleSignals.SetSignalValue(mmsXSignal, cameraHorizontalOrientation);
		m_VehicleSignals.SetSignalValue(mmsYSignal, cameraVerticalOrientation);
		camera.SetYawPitchRoll(Vector(-cameraHorizontalOrientation, cameraVerticalOrientation, 0));


		if (enableDebug)
		{
			////// DEBUG LINES /////////
			m_aDebugShapes.Clear();
			Shape forwardShape = Shape.Create(ShapeType.LINE, ARGB(255, 0, 255, 0), ShapeFlags.NOZBUFFER, mmsStart, mmsStart + mmsDirection * 1550);
			m_aDebugShapes.Insert(forwardShape);

			Shape desiredMmsDirShape = Shape.Create(ShapeType.LINE, ARGB(255, 0, 255, 0), ShapeFlags.NOZBUFFER,  mmsStart, targetPos);
			m_aDebugShapes.Insert(desiredMmsDirShape);

			vector mmsVector = camera.GetYawPitchRoll().AnglesToVector();
			Shape mmsDirShape = Shape.Create(ShapeType.LINE, ARGB(255, 0, 0, 0), ShapeFlags.NOZBUFFER,  mmsStart, mmsStart + mmsDirection * 1550);
			m_aDebugShapes.Insert(mmsDirShape);
			////// DEBUG LINES /////////
		}
	}

	protected vector GetYawPitchRoll(vector vectorA, vector vectorB)
    {
        vector normA = vectorA.Normalized();
        vector normB = vectorB.Normalized();
        float yaw = Math.Atan2(normB[0], normB[2]) - Math.Atan2(normA[0], normA[2]);
        float pitch = Math.Asin(normB[1]) - Math.Asin(normA[1]);
        vector cross = normA * normB;
        float roll = Math.Atan2(cross[1], cross[2]);
        yaw = -yaw * Math.RAD2DEG;
        pitch = -pitch * Math.RAD2DEG;
        roll = roll * Math.RAD2DEG;
        return Vector(yaw, pitch, roll);
    }

	private void UpdateMMSVectors()
	{
		vehicle.GetBoneMatrix(mmsPos, mmsStartMat);
		vehicle.GetBoneMatrix(mmsDir, mmsEndMat);
		vehicle.GetBoneMatrix(aircraftFWD, aircraftMat);
		mmsStart = vehicle.CoordToParent(mmsStartMat[3]);
		mmsEnd = vehicle.CoordToParent(mmsEndMat[3]);
		aircraftFWDPos = vehicle.CoordToParent(aircraftMat[3]);
	}

	private RaycastResult RayCastFromCamera(vector direction)
	{
		autoptr TraceParam param = new TraceParam;
		param.LayerMask = EPhysicsLayerDefs.Projectile;
		param.Exclude = vehicle;
		param.Flags = TraceFlags.ENTS | TraceFlags.VISIBILITY | TraceFlags.WORLD | TraceFlags.OCEAN;
		param.Start = mmsStart + (direction * 15);
		param.End = mmsStart + (direction * maxRaycastDistanceForCameraStabilization);

		auto distance = GetGame().GetWorld().TraceMove(param, null);
		
		return new RaycastResult(distance, param.TraceEnt);
	}

	private void StartCamera()
	{
		if(!vehicle)
			return;
		
		if (!IsVehicleCrew())
			return;

		TNodeId boneID = vehicle.GetAnimation().GetBoneIndex("v_mms_dir");
		vector mat[4];
		vehicle.GetAnimation().GetBoneMatrix(boneID, mat);


		float viewDistance = GetGame().GetViewDistance();
		camera = CreatePIPCameraOld(vehicle, mat[3], cameraIndex, cameraFOV1, 1, 8000);
		
		laserDesignator.SetCameraEntity(camera);
		if (!camera)
		{
			Print("Could not create PIP camera!", LogLevel.ERROR);
			return;
		}
		flirUtil.SetCameraOverlay(CameraVisionMode.TV, cameraIndex);
	}

	// ScriptCamera is faster, updates run faster
	private ScriptCamera CreatePIPCameraOld(IEntity parent, vector position, int index, float fov, float nearPlane, float farPlane)
	{
		// Spawn camera
		BaseWorld baseWorld = parent.GetWorld();
		ScriptCamera pipCamera = ScriptCamera.Cast(GetGame().SpawnEntity(ScriptCamera, baseWorld));

		vector mat[4];
		Math3D.MatrixIdentity3(mat);
		mat[3] = position;


		pipCamera.SetTransform(mat);
		pipCamera.Index = index;
		pipCamera.FreeFly = false;

		baseWorld.SetCameraVerticalFOV(index, fov);
		baseWorld.SetCameraFarPlane(index, farPlane);
		baseWorld.SetCameraNearPlane(index, nearPlane);
		baseWorld.SetCameraType(index, CameraType.PERSPECTIVE);

		parent.AddChild(pipCamera, -1, EAddChildFlags.AUTO_TRANSFORM);
		return pipCamera;
	}
	
	private SCR_PIPCamera CreatePIPCamera(IEntity parent, vector position, int index, float fov, float nearPlane, float farPlane)
	{
		// Spawn camera
		BaseWorld baseWorld = parent.GetWorld();
		EntitySpawnParams params = new EntitySpawnParams();
		SCR_PIPCamera pipCamera = SCR_PIPCamera.Cast(GetGame().SpawnEntity(SCR_PIPCamera, baseWorld, params));

		vector mat[4];
		Math3D.MatrixIdentity3(mat);
		mat[3] = position;


		pipCamera.SetTransform(mat);
		pipCamera.SetCameraIndex(index);
		pipCamera.SetFOVDegree(fov);
		pipCamera.SetFarPlane(farPlane);
		pipCamera.SetNearPlane(nearPlane);
		parent.AddChild(pipCamera, -1, EAddChildFlags.AUTO_TRANSFORM);
		pipCamera.UpdatePIPCamera(1.0);
		return pipCamera;
	}

	void ToggleOverlay()
	{
		if (!IsVehicleCrew())
			return;
		
		visionMode++;
		if (visionMode > 2)
		{
			visionMode = 0;
		}

		switch (visionMode)
		{
			case 0: flirUtil.SetCameraOverlay(CameraVisionMode.TV, cameraIndex); break;
			case 1: flirUtil.SetCameraOverlay(CameraVisionMode.WHOT, cameraIndex); break;
			case 2: flirUtil.SetCameraOverlay(CameraVisionMode.BHOT, cameraIndex); break;
		}
	}

	bool IsVehicleCrew()
	{
		if (!GetGame().GetPlayerController())
			return false;
		
		IEntity player = GetGame().GetPlayerController().GetControlledEntity();
		if (!player)
			return false;
		
		IEntity playerVehicle = player.GetParent();
		IEntity vehicleEnt = GetOwner().GetParent();
		
		if(!playerVehicle)
			return false;
		
		if(!vehicle)
			return false;
		
		return playerVehicle.GetID() == vehicleEnt.GetID();
	}
	
	private float GetLimitAdjustedVerticalOrientation(float angle)
	{
		if(angle > 0)
			return Math.Min(angle, cameraVerticalUpLimit);
		else 
			return Math.Max(angle, cameraVerticalDownLimit);
	}

		RplId creatorId = RplComponent.Cast(player.FindComponent(RplComponent)).Id();
       	SCR_ChimeraCharacter character = SCR_ChimeraCharacter.Cast(player);

	}
