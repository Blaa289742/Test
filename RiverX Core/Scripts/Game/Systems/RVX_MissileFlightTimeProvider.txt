class RVX_MissileFlightTimeProviderClass : ScriptGameComponentClass
{

}

class RVX_MissileFlightTimeProvider : ScriptGameComponent
{
	protected WCS_Armament_LoadoutComponent loadoutComponent;
	protected RVX_LaserDesignatorComponent laserDesignator;
	protected Projectile launchedProjectile;
	protected WCS_Armament_WeaponComponent currentWeapon;
	
	//Launched projectile components
	protected RVX_AGM_WCS_Armament_LaserSeekerComponent projectileSeeker;
	
	//Next projectile components
	protected RVX_AGM_WCS_Armament_LaserSeekerComponent nextProjectileSeeker;


	override void OnPostInit(IEntity owner)
    {
        super.OnPostInit(owner);
		loadoutComponent = WCS_Armament_LoadoutComponent.Cast(owner.FindComponent(WCS_Armament_LoadoutComponent));
		GetGame().GetCallqueue().CallLater(Init, 1000, false);
    }
	
	void Init()
	{
		laserDesignator = Inject<RVX_LaserDesignatorComponent>.Get(GetOwner());
		if (!laserDesignator)
		{
			laserDesignator = Inject<RVX_LaserDesignatorComponent>.FindInAnySlot(GetOwner());
		}
		loadoutComponent.GetWCSWeaponManager().GetOnWeaponChanged().Insert(OnWeaponChanged);
		WCS_Armament_WeaponComponent weapon = WCS_WeaponQueue.Cast(loadoutComponent.GetCurrentWeaponQueue()).GetCurrentWeapon();
		OnWeaponChanged(weapon);
	}
	
	float GetMissileFlightTime()
	{
		/*if(GetCurrentWeaponType() != RVXWeaponType.AGMLaser)
			return 0;*/
		
		WCS_WeaponQueue queue = WCS_WeaponQueue.Cast(loadoutComponent.GetCurrentWeaponQueue());
		if (!queue)
			return 0;
		
		if (!queue.GetCurrentWeapon())
			return 0;
		
		
		if (projectileSeeker)
		{
			float targetDist = projectileSeeker.GetTargetDistance();
			float missileSpeed = launchedProjectile.GetPhysics().GetVelocity().Length();
			if (missileSpeed == 0)
				return 0;
			float result = Math.Round(targetDist / (missileSpeed * 1.9) * 10) / 10;
			return result;
		} else if(nextProjectileSeeker && laserDesignator.IsLaserActive()) {
			
			float laserDist = laserDesignator.GetLaserDistance();
			float missileSpeed = nextProjectileSeeker.GetRatedMissileSpeed();
			if (missileSpeed == 0)
				return 0;
			float result = Math.Round(laserDist / (missileSpeed * 1.9) * 10) / 10;
			return result;
		}
		
		return 0;
	}
	
	bool IsMissileLaunched()
	{
		return launchedProjectile != null;
	}

	private void OnWeaponChanged(WCS_Armament_WeaponComponent weapon)
	{
		weapon.GetOnProjectileLaunched().Remove(OnProjectileLaunched);
		weapon.GetOnProjectileLaunched().Insert(OnProjectileLaunched);
		currentWeapon = weapon;
		Projectile nextProjectile = currentWeapon.GetNextProjectile();
		if (nextProjectile)
			nextProjectileSeeker = RVX_AGM_WCS_Armament_LaserSeekerComponent.Cast(nextProjectile.FindComponent(RVX_AGM_WCS_Armament_LaserSeekerComponent));
		else
			nextProjectileSeeker = null;
	}
	
	private void OnProjectileLaunched(Projectile projectile)
	{
		launchedProjectile = projectile;
		if(!launchedProjectile)
			return;
		projectileSeeker = RVX_AGM_WCS_Armament_LaserSeekerComponent.Cast(launchedProjectile.FindComponent(RVX_AGM_WCS_Armament_LaserSeekerComponent));
		Projectile nextProjectile = currentWeapon.GetNextProjectile();
		if (nextProjectile)
			nextProjectileSeeker = RVX_AGM_WCS_Armament_LaserSeekerComponent.Cast(nextProjectile.FindComponent(RVX_AGM_WCS_Armament_LaserSeekerComponent));
		else
			nextProjectileSeeker = null;
	}
}
