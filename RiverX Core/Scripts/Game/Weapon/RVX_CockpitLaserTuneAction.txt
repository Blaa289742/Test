class RVX_CockpitLaserTuneAction : ScriptedUserAction
{
	protected RVX_LaserSeeker laserSeekerComp;
	
	//------------------------------------------------------------------------------------------------
	[Attribute("0")]
	protected int laserCodeDigit;
	
	override void Init(IEntity pOwnerEntity, GenericComponent pManagerComponent)
	{
		laserSeekerComp = RVX_LaserSeeker.Cast(pOwnerEntity.FindComponent(RVX_LaserSeeker));
	}
	
	override void PerformAction(IEntity pOwnerEntity, IEntity pUserEntity)
	{
		string code = laserSeekerComp.GetCode();
		array<string> digits = {};
		
		digits.Insert(code.Get(0));
		digits.Insert(code.Get(1));
		digits.Insert(code.Get(2));
		digits.Insert(code.Get(3));
		
		// set code
		int digit = digits.Get(laserCodeDigit).ToInt();
		digit++;
		if (digit > 8)
		{
			digit = 1;
		}
		digits.Set(laserCodeDigit, digit.ToString());
		string resultCode = SCR_StringHelper.Join("", digits, true);
		laserSeekerComp.SetLaserCode(resultCode);
	}
	
	override bool HasLocalEffectOnlyScript()
	{
		return false;
	}
	
	override bool GetActionNameScript(out string outName)
	{
		if (!laserSeekerComp)
			return false;

		string code = laserSeekerComp.GetCode();	

		outName = "LSS code [" + laserCodeDigit + "]: " + code.Get(laserCodeDigit);
		return true;
	}
}
