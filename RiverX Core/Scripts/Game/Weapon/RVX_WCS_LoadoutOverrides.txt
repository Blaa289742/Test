modded class WCS_Armament_LoadoutComponent
{
	private ref ScriptInvoker weaponLoadoutChangeInvoker = new ScriptInvoker();
	
	protected void CreateWeaponQueueFast()
	{
		if (m_RplComponent.IsProxy())
			return;

		array<RplId> weaponsIds = {};
		m_WeaponManager.LoadWeapons(m_ArmamentGroups, GetOwner(), weaponsIds);

		GetGame().GetCallqueue().CallLater(CreateWeaponQueueClient, 100, false, weaponsIds);
	}
	
	WCS_WeaponManager GetWCSWeaponManager() 
	{
		return m_WeaponManager;
	}
	
	ScriptInvoker OnLoadoutChanged()
	{
		return weaponLoadoutChangeInvoker;
	}
	
	protected void WeaponsChanged()
	{
		weaponLoadoutChangeInvoker.Invoke();
	}
}

modded class WCS_WeaponManager
{
	ref ScriptInvoker<int> currentWeaponSlotChanged = new ScriptInvoker<int>();
	ref ScriptInvokerVoid weaponFired = new ScriptInvokerVoid();
	
	int GetDifferentWeaponsCount()
	{
		return m_aWeaponQueues.Count();
	}
	
	override void CycleWeaponGroup(int currentWeaponManagerSlotIndex)
	{
		super.CycleWeaponGroup(currentWeaponManagerSlotIndex);
		currentWeaponSlotChanged.Invoke(currentWeaponManagerSlotIndex);
	}
	
	override void Fire(vector parentVelocity, IEntity gunner)
	{
		super.Fire(parentVelocity, gunner);
		weaponFired.Invoke();
	}
	
	ScriptInvoker<int> OnWeaponSlotIndexChanged()
	{
		return currentWeaponSlotChanged;
	}

	
	ScriptInvokerVoid GetOnWeaponFired()
	{
		return weaponFired;
	}
}

modded class WCS_Armament_MissileMovementComponent
{
    [Attribute("0", category: "Debug", desc: "Debug draw flight path")]
	protected bool debugDrawFlightPath;
	
	protected static ref array<ref Shape> debugShapes = {};
	private vector previousPos;
	
	override void MoveTowardsTarget(float timeSlice, vector targetPosition)
    {
        super.MoveTowardsTarget(timeSlice, targetPosition);
		
		IEntity owner = GetOwner();
		if (debugDrawFlightPath && previousPos)
		{	
			Shape line = Shape.Create(ShapeType.LINE, ARGB(255, 0, 255, 0), ShapeFlags.NOZBUFFER, previousPos, owner.GetOrigin());
			WCS_Armament_MissileMovementComponent.debugShapes.Insert(line);
		}
		
		previousPos = owner.GetOrigin();
    }
}
