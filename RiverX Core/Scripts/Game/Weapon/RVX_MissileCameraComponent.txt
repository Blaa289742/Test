class RVX_MissileCameraComponentClass :  ScriptComponentClass {}

class RVX_MissileCameraComponent : ScriptComponent
{
	[Attribute("{302BBC03EADC289C}Prefabs/Editor/Camera/RVX_ManualCameraPhoto.et", UIWidgets.ResourceNamePicker, "")]	
	private ResourceName cameraPrefab;		
	
	private Projectile currentProjectile;
	private bool isAttachedToMissile = false;
	private SCR_ManualCamera misileCamera;
	private WCS_Armament_LoadoutComponent loadoutComponent;
	private IEntity ownerEnt;
	
	
	override void OnPostInit(IEntity owner)
    {
       	ownerEnt = owner;
		//GetGame().GetCallqueue().CallLater(RegisterHandler, 1000, false, owner);
    }
	
	void Register()
	{
		loadoutComponent = Inject<WCS_Armament_LoadoutComponent>.Get(ownerEnt);
		
		GetGame().GetInputManager().AddActionListener("MissleCameraToggle", EActionTrigger.DOWN, OnToggleMissileCamera);
		
		loadoutComponent.OnLoadoutChanged().Insert(AssignProjectileLaunchedToCurrentWeapons);
		AssignProjectileLaunchedToCurrentWeapons();
	}
	
	void Unregister()
	{
		loadoutComponent = Inject<WCS_Armament_LoadoutComponent>.Get(ownerEnt);
		
		GetGame().GetInputManager().RemoveActionListener("MissleCameraToggle", EActionTrigger.DOWN, OnToggleMissileCamera);
		loadoutComponent.OnLoadoutChanged().Remove(AssignProjectileLaunchedToCurrentWeapons);
		
		RemoveAsignProjectileLaunchedToCurrentWeapons();
	}
	
	private void OnToggleMissileCamera()
	{
		if (currentProjectile)
		{
			if (misileCamera)
			{
				ReturnCameraToPlayer();
			} else
			{
				AttachCameraToMissile(currentProjectile);
			}
		} else 
		{
			ReturnCameraToPlayer();
		}
	}
	
	private void AssignProjectileLaunchedToCurrentWeapons()
	{
		array<WCS_Armament_WeaponComponent> weapons = loadoutComponent.GetAllWeapons();
		foreach(WCS_Armament_WeaponComponent weapon: weapons)
		{
			weapon.GetOnProjectileLaunched().Remove(OnProjectileLaunched);
			weapon.GetOnProjectileLaunched().Insert(OnProjectileLaunched);
		}
	}
	
	private void RemoveAsignProjectileLaunchedToCurrentWeapons()
	{
		array<WCS_Armament_WeaponComponent> weapons = loadoutComponent.GetAllWeapons();
		foreach(WCS_Armament_WeaponComponent weapon: weapons)
		{
			weapon.GetOnProjectileLaunched().Remove(OnProjectileLaunched);
		}
	}
	
	private void OnProjectileLaunched(Projectile projectile)
	{
		currentProjectile = projectile;
	}

	private void AttachCameraToMissile(Projectile projectile)
	{
		CameraManager cameraManager = GetGame().GetCameraManager();
		IEntity camera = GetGame().SpawnEntityPrefab(Resource.Load(cameraPrefab));        
		if(!camera)
		        return;  
		projectile.AddChild(camera, -1, EAddChildFlags.AUTO_TRANSFORM);
		
		
		// Place camera close to missile
		vector mat[4];
		projectile.GetTransform(mat);
		vector offset = "5 0 -5";
		mat[3] = projectile.CoordToParent(offset);
		camera.SetTransform(mat);
		
		// Rotate towards
		vector transform[4];
		camera.GetTransform(transform);

		vector rotationMatrix[4];
		SCR_Math3D.LookAt(camera.GetOrigin(), projectile.GetOrigin(), vector.Up, rotationMatrix);

		float quat[4];
		Math3D.MatrixToQuat(rotationMatrix, quat);

		vector currentAngles = camera.GetYawPitchRoll();
		vector angles = Math3D.QuatToAngles(quat);

		Math3D.AnglesToMatrix(angles, transform);
		camera.SetTransform(transform);
		
		
		misileCamera = SCR_ManualCamera.Cast(camera);
		if(!misileCamera)
		        return; 
		// Set camera target
		RVX_OrbitingManualCameraComponent orbitComponent = RVX_OrbitingManualCameraComponent.Cast(misileCamera.FindCameraComponent(RVX_OrbitingManualCameraComponent));
		orbitComponent.SetTargetEntity(projectile);
		
		//originalCamera = cameraManager.CurrentCamera();
		
		// Set current camera to projectile cam
		cameraManager = GetGame().GetCameraManager();
		cameraManager.SetCamera(misileCamera);
	}
	
	private void ReturnCameraToPlayer()
	{
		SCR_EntityHelper.DeleteEntityAndChildren(misileCamera);
	}
}
