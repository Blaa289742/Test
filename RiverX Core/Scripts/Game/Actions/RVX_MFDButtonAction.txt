class RVX_MFDButtonAction : ScriptedUserAction
{
	[Attribute(defvalue: "T1", desc: "Index of the MFD button (T1, T2 etc.")]
	protected string buttonIndex;
	
	//------------------------------------------------------------------------------------------------
	override void PerformAction(IEntity pOwnerEntity, IEntity pUserEntity)
	{		
		RVX_MFD_Manager mfdManager = RVX_MFD_Manager.Cast(pOwnerEntity.FindComponent(RVX_MFD_Manager));
		mfdManager.OnButtonPressed(buttonIndex);
	}

	//------------------------------------------------------------------------------------------------
	override bool CanBeShownScript(IEntity user)
	{
		return true;
	}

	//------------------------------------------------------------------------------------------------
	override bool CanBePerformedScript(IEntity user)
	{
		return true;
	}

	//------------------------------------------------------------------------------------------------
	override bool HasLocalEffectOnlyScript()
	{
		return false;
	}
}
