class RVX_LaserSeekerClass : ScriptGameComponentClass
{

}

class RVX_LaserSeeker : ScriptGameComponent
{	
	[Attribute("false", UIWidgets.Auto, "Enable debug lines and print", "Debug")]
    protected bool enableDebug;
	
	[Attribute("35", UIWidgets.Auto, "Horizontal limit for detecting laser", "Debug")]
    protected float horizontalDetectionAngleLimit;
	
	[Attribute("35", UIWidgets.Auto, "Vertical limit for detecting laser", "Debug")]
    protected float verticalDetectionAngleLimit;
	
	[Attribute("40000", UIWidgets.Auto, "Laser detection distance limit", "Debug")]
    protected float laserDetectionRangeLimit;
	
	[Attribute("", UIWidgets.Auto, "Laser seeker origin")]
    ref protected PointInfo laserSeekerOriginPos;
	
	[Attribute("AGM", UIWidgets.Auto, "Laser seeker ID")]
    protected string laserSeekerID;
	
	protected IEntity seekerEntity;
	protected vector laserTargetPosition;
	protected vector laserSeekerStart;
	protected bool seekerActive;
	protected bool laserTargetFound = false;
	protected string laserCode = "1688";

	
	protected vector mmsStartMat[4];
	protected ref array<ref Shape> m_aDebugShapes = {};
	
	protected float m_fTargetCheckInterval = (1000/30) * 0.001; // 30fps
	protected float m_fTargetCheckIntervalTimer = float.MAX;
	
	override void EOnInit(IEntity owner)
	{
		super.EOnInit(owner);
		laserSeekerOriginPos.Init(owner);
	}
	
	void ActivateSearch()
	{
		seekerActive = true;
		SendRemoteUpdate();
		SetEventMask(seekerEntity, EntityEvent.POSTFRAME);
	}
	
	void DeactivateSearch()
	{
		seekerActive = false;
		laserTargetFound = false;
		SendRemoteUpdate();
		ClearEventMask(seekerEntity, EntityEvent.POSTFRAME);
	}
	
	bool IsLaserTargetFound()
	{
		return laserTargetFound;
	}
	
	bool IsSeekerActive()
	{
		return seekerActive;
	}
	
	string GetCode()
	{
		return laserCode;
	}
	
	void SetLaserCode(string code)
	{
		laserCode = code;
	}
	
	vector GetDirectionToLaserTarget(bool checkConstraints = true)
	{
		if(!laserTargetFound)
			return Vector(0, 0 , 0);
		
		vector rotationMatrix[4];
		SCR_Math3D.LookAt(laserSeekerStart, laserTargetPosition, vector.Up, rotationMatrix);

		float quat[4];
		Math3D.MatrixToQuat(rotationMatrix, quat);

		vector angles = Math3D.QuatToAngles(quat);	
		vector seekerAngles = ToSeekerAngles(angles);
		if (checkConstraints && (Math.AbsFloat(seekerAngles[0]) > horizontalDetectionAngleLimit || Math.AbsFloat(seekerAngles[1]) > verticalDetectionAngleLimit))
			return Vector(0, 0 , 0);
		 
		return angles;
	}
	
	vector GetLaserTargetPosition()
	{
		return laserTargetPosition;
	}
	
	vector GetSeekerAngles()
	{
		vector targetDir = GetDirectionToLaserTarget(false);
		return ToSeekerAngles(targetDir);
	}
	
	vector GetLaserSeekerOrigin()
	{
		return laserSeekerStart;
	}
	
	string GetLaserSeekerID()
	{
		return laserSeekerID;
	}
	
	bool IsTargetAnglesWithinConstraints()
	{
		vector seekerAngles = GetSeekerAngles();
		return Math.AbsFloat(seekerAngles[0]) < horizontalDetectionAngleLimit && Math.AbsFloat(seekerAngles[1]) < verticalDetectionAngleLimit;
	}
	
	override void OnPostInit(IEntity owner)
	{
		super.OnPostInit(owner);
		seekerEntity = owner;
	}
	
	override event void EOnPostFrame(IEntity owner, float timeSlice)
	{
		m_fTargetCheckIntervalTimer += timeSlice;
		if (m_fTargetCheckIntervalTimer < m_fTargetCheckInterval)
			return;
		m_fTargetCheckIntervalTimer = 0;
		
		UpdateSeekerVectors();
		UpdateTarget(owner);
		
		
		if (laserTargetFound && enableDebug)
		{
			////// DEBUG LINES /////////
			m_aDebugShapes.Clear();
			Shape forwardShape = Shape.Create(ShapeType.LINE, ARGB(255, 0, 255, 0), ShapeFlags.NOZBUFFER, laserSeekerStart, laserSeekerStart + GetDirectionToLaserTarget().AnglesToVector() * 1550);
			m_aDebugShapes.Insert(forwardShape);

			Shape desiredMmsDirShape = Shape.Create(ShapeType.LINE, ARGB(255, 0, 255, 0), ShapeFlags.NOZBUFFER,  laserSeekerStart, laserTargetPosition);
			m_aDebugShapes.Insert(desiredMmsDirShape);
			////// DEBUG LINES /////////
		}
	}
	
	private void UpdateTarget(IEntity owner)
	{
		ref LaserTargetData laserTarget = GetClosestLaserTarget();
		if (!laserTarget)
		{
			laserTargetFound = false;
		} else 
		{
			laserTargetPosition = laserTarget.GetPosition();
			laserTargetFound = true;
		}
	}
	
	ref LaserTargetData GetClosestLaserTarget()
	{		
		array<ref LaserTargetData> laserTargets = RVX_LaserTargetManager.getInstance().GetLaserTargets(GetCode());
		ref LaserTargetData closestLaserTarget;
		
		foreach (LaserTargetData laserTarget: laserTargets)
		{
			if (!closestLaserTarget || vector.DistanceSq(GetOwner().GetOrigin(), laserTarget.GetPosition()) < vector.DistanceSq(GetOwner().GetOrigin(), closestLaserTarget.GetPosition()))
			{
				closestLaserTarget = laserTarget;
			}
		}
		return closestLaserTarget;
	}
	
	private vector ToSeekerAngles(vector targetDir)
	{
		vector diff = seekerEntity.GetYawPitchRoll() - targetDir;
		return Vector(-diff[0], -diff[1], 0);
	}
	
	private void UpdateSeekerVectors()
	{
		laserSeekerStart = seekerEntity.GetOrigin();
		
		vector mat[4];
		laserSeekerOriginPos.Init(seekerEntity);
		laserSeekerOriginPos.GetWorldTransform(mat);
		laserSeekerStart = mat[3];
	}

	
	//-------------------------------------------------------------------------------------
	void OnReceivedLaserSeekerUpdate(bool state, string laserCode)
	{
		if (state)
		{
			ActivateSearch();
		} else 
		{
			DeactivateSearch();
		}
		SetLaserCode(laserCode);
	}
	
	void SendRemoteUpdate()
	{
		if (Replication.IsServer())
		{
			return;
		}
		IEntity player = GetGame().GetPlayerController().GetControlledEntity();
		RplId creatorId = RplComponent.Cast(GetOwner().FindComponent(RplComponent)).Id();
       	SCR_ChimeraCharacter character = SCR_ChimeraCharacter.Cast(player);
		character.ToggleLaserSeeker(creatorId, laserSeekerID, seekerActive, laserCode);
	}
	
	static RVX_LaserSeeker FindLaserSeeekerByID(IEntity entity, string id) {
		RVX_LaserSeeker foundComponent;
		ref array<Managed> comps = {};
		entity.FindComponents(RVX_LaserSeeker, comps);
		
		foreach(Managed comp: comps)
		{
			RVX_LaserSeeker seeker = RVX_LaserSeeker.Cast(comp);
			if (seeker && seeker.GetLaserSeekerID() == id)
				return seeker;
		}
		return null;
	}
}

modded class SCR_ChimeraCharacter
{ 		
	void ToggleLaserSeeker(RplId helicopterId, string laserSeekerID, bool state, string laserCode)
    { 
        Rpc(RpcAsk_ToggleLaserSeeker, helicopterId, laserSeekerID, state, laserCode);
    }
	
	[RplRpc(RplChannel.Reliable, RplRcver.Server)]
    private void RpcAsk_ToggleLaserSeeker(RplId helicopterId, string laserSeekerID, bool state, string laserCode)
    {
		RpcAsk_BroadcastToggleLaserSeeker(helicopterId, laserSeekerID, state, laserCode);
		//Rpc(RpcAsk_BroadcastToggleLaserSeeker, helicopterId, laserSeekerID, state, laserCode);
    }
	
	[RplRpc(RplChannel.Reliable, RplRcver.Broadcast)]
    private void RpcAsk_BroadcastToggleLaserSeeker(RplId helicopterId, string laserSeekerID, bool state, string laserCode)
    {
       	RplComponent comp = RplComponent.Cast(Replication.FindItem(helicopterId));
		if(comp)
		{
			IEntity vehicle = comp.GetEntity();
			array<Managed> seekers = {};
			vehicle.FindComponents(RVX_LaserSeeker, seekers);
			foreach(Managed seeker: seekers)
			{
				if (RVX_LaserSeeker.Cast(seeker).GetLaserSeekerID() == laserSeekerID)
				{
					RVX_LaserSeeker.Cast(seeker).OnReceivedLaserSeekerUpdate(state, laserCode);
				}
			}
		}
    }
}
