class RVX_DraggableWidget: ScriptedWidgetComponent
{
	[Attribute()]
	int myInt;
	
	string waypointID;
	
	private bool isDragging = false;
	private Widget element;
	private float width;
	private float height;
	private float percentPosX;
	private float percentPosY;
	
	override bool OnMouseButtonDown(Widget w, int x, int y, int button)
	{
		isDragging = true;
		element = w;
		width = FrameSlot.GetSizeX(element);
		height = FrameSlot.GetSizeY(element);
		
		return true;
	}
	
	override bool OnMouseButtonUp(Widget w, int x, int y, int button)
	{
		isDragging = false;
		return true;
	}
	
	void OnMouseMove(int x, int y)
	{
		if (element && isDragging)
		{
			FrameSlot.SetPos(element, x - width/2, y - height/2);
			percentPosX = x / FrameSlot.GetSizeX(element.GetParent());
			percentPosY = y / FrameSlot.GetSizeY(element.GetParent());
		}
	}
	
	void SetPosition(int x, int y)
	{
		if (element)
		{
			FrameSlot.SetPos(element, x - width/2, y - height/2);
			percentPosX = x / FrameSlot.GetSizeX(element.GetParent());
			percentPosY = y / FrameSlot.GetSizeY(element.GetParent());
		}
	}
	
	void OnParentResized()
	{
		if (element)
		{
			float resizedX = FrameSlot.GetSizeX(element.GetParent()) * percentPosX;
			float resizedY = FrameSlot.GetSizeY(element.GetParent()) * percentPosY;
			FrameSlot.SetPos(element, resizedX - width/2, resizedY - height/2);
		}
	}
	
	void Create(Widget widget, int w, int h, int x, int y, string wptID)
	{
		element = widget;
		width = w;
		height = h;
		FrameSlot.SetSize(element, w, h);
		FrameSlot.SetPos(element, x - w/2, y - h/2);
		waypointID = wptID;
		percentPosX = x / FrameSlot.GetSizeX(element.GetParent());
		percentPosY = y / FrameSlot.GetSizeY(element.GetParent());
		
		TextWidget.Cast(widget.FindAnyWidget("WaypointID")).SetText(wptID);
	}
	
	float GetPosX()
	{
		return FrameSlot.GetPosX(element) + width / 2;
	}
	
	float GetPosY()
	{
		return FrameSlot.GetPosY(element) + height / 2;
	}
	
	Widget GetWidget()
	{
		return element;
	}
	
	string GetWaypointID()
	{
		return waypointID;
	}
}
