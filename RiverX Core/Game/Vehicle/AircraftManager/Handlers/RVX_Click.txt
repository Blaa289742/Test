class RVX_Click: ScriptedWidgetComponent
{
	private ref ScriptInvoker action = new ScriptInvoker();
	private string handlerParam = "";
	
	override bool OnClick(Widget w, int x, int y, int button)
	{
		action.Invoke(handlerParam);
		return true;
	}
	
	ScriptInvoker GetOnClick(string param = "")
	{
		// passing data to handle
		handlerParam = param; 
		return action;
	}
}
