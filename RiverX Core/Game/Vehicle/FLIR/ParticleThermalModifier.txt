/*
Currently particles have neutral temperature and are not visible in Thermal. This class changes it, and should be removed when this behaviour changes with BI updates
*/
class ParticleThermalModifier
{
	private static ref array<ResourceName> resources = {};
	private static ref array<ref Material> materials = {};
	static void AdjustParticleThermals()
	{
		if (ParticleThermalModifier.resources.Count() < 1)
		{
			SearchResourcesFilter filter = new SearchResourcesFilter();
			filter.fileExtensions = {"emat"};
			ResourceDatabase.SearchResources(filter, ParticleThermalModifier.resources.Insert); 
			
			foreach(ResourceName resource: ParticleThermalModifier.resources)
			{
				string resourceLower = resource;
				resourceLower.ToLower();
				if(!resourceLower.Contains("smoke"))
				{
					continue;
				}
				materials.Insert(Material.GetOrLoadMaterial(resource, 0));
			}
		}
		
		foreach(Material material: ParticleThermalModifier.materials)
		{
			AdjustParticleMaterialThermal(material);
		}
		
	}
	
	private static void AdjustParticleMaterialThermal(Material mat)
	{
		float tex[4];
		tex[0] = 1;
		tex[1] = 1;
		tex[2] = 1;
		tex[3] = 1;
		mat.SetParam("EmissiveLV", -12);
		mat.SetParam("Emissive", tex);
		mat.SetParam("TIVisibility", 2);
		mat.SetParam("TIBlendMode", 1);
		mat.SetParam("Temperature", 1000);
		mat.SetParam("TempCoefMult", 10);
		mat.SetParam("EmissionToTempCoef", 10);
	}
}
