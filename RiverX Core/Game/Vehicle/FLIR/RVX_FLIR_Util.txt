enum CameraVisionMode
{
	TV,
	BHOT,
	WHOT
}

class RVX_FLIR_Util 
{
	// resource
	private static ResourceName flirOverlayWHOTResource = "{DB95278D774CFD33}Assets/FLIR/ThermalEffect.emat";
	private static ResourceName flirOverlayBHOTResource = "{B397611889C8DD69}Assets/FLIR/ThermalEffect_Inverted.emat";
	private static ResourceName tvOverlayResource = "{25F5518F7DF79A8F}Assets/FLIR/PostProcess_TV.emat";
	private static ResourceName cameraOverlayResource;

	// misc
	private static IEntity ownerEnt;
	private static ref map<int, CameraVisionMode> cameras = new map<int, CameraVisionMode>;
	private static float entityScanTimer = 0.0;
	private static float particleUpdateTimer = 0.0;
	private static const float ENTITY_SCAN_INTERVAL = 0.5; // 10 times per second
	private static const float PARTICLE_UPDATE_INTERVAL = 1.0; // once per second

	private static ref array<ResourceName> modifiedResources = {};
	private static ref map<EntityID, IEntity> modifiedEntities = new map<EntityID, IEntity>();

	// consts
	private static float RADIUS = 4500;

	// ---------------- Singleton----------------------
	private static ref RVX_FLIR_Util instance;

    static RVX_FLIR_Util getInstance() {
        if (instance == null) {
            instance = new RVX_FLIR_Util();
        }
        return instance;
    }
	// ---------------- Singleton----------------------

	// Called each frame to set brightness and other material params
	void UpdateCamera(float timeSlice)
	{
		UpdateCameraBrightness();
		Update(timeSlice);
	}

	void SetFLIROwnerEntity(IEntity entity)
	{
		ownerEnt = entity;
	}

	void SetCameraID(int camId)
	{
		cameras.Insert(camId, CameraVisionMode.TV);
	}

	// Called when vision mode of camera is changed
	void SetCameraOverlay(CameraVisionMode overlay, int camID)
	{
		if (overlay == CameraVisionMode.TV) {
			UpdateCameraBrightness();
			GetGame().GetWorld().SetCameraPostProcessEffect(camID, 10, PostProcessEffectType.HDR, tvOverlayResource);
			cameras.Set(camID, CameraVisionMode.TV);
		}
		if (overlay == CameraVisionMode.WHOT) {
			GetGame().GetWorld().SetCameraPostProcessEffect(camID, 10, PostProcessEffectType.ThermalImaging, flirOverlayWHOTResource);
			cameras.Set(camID, CameraVisionMode.WHOT);
		}
		if (overlay == CameraVisionMode.BHOT) {
			GetGame().GetWorld().SetCameraPostProcessEffect(camID, 10, PostProcessEffectType.ThermalImaging, flirOverlayBHOTResource);
			cameras.Set(camID, CameraVisionMode.BHOT);
		}

		Update(0);
	}

	void ClearCache()
	{
		modifiedResources.Clear();
		modifiedEntities.Clear();
	}

	// Unused in current impl
	void SetTargetThermalOff() {}

	private void UpdateCameraBrightness()
	{
	    float mainCameraHDRBrightness = GetGame().GetWorld().GetCameraHDRBrightness(0);

		foreach(int camID, CameraVisionMode visionMode: cameras)
		{
			if (visionMode == CameraVisionMode.TV)
			{
				GetGame().GetWorld().SetCameraHDRBrightness(camID, mainCameraHDRBrightness);
			}
		}
	}

	private void Update(float timeSlice)
	{
        entityScanTimer += timeSlice;
	    particleUpdateTimer += timeSlice;

		// Run particle thermal update once per second
	    if (particleUpdateTimer >= PARTICLE_UPDATE_INTERVAL)
	    {
	        particleUpdateTimer = 0.0;
			UpdateParticles();
	    }

		// Run entity scan 10 times per second
	    if (entityScanTimer >= ENTITY_SCAN_INTERVAL)
	    {
	        entityScanTimer = 0.0;
			UpdateNearEntities();
		}
	}
	
	float lastScanTime = 0;
	void UpdateMaterialsThermals()
	{
		if (System.GetTickCount() - lastScanTime < 5000)
			return;
		
		array<ResourceName> resources = EntityMaterialsUtil.GeMaterials();
		foreach(ResourceName resource: resources)
		{
			AdjustMaterialTemperature(resource);
		}
	}
	
	void UpdateMaterialsThermalsForCharacter(IEntity entity)
	{	
		array<ResourceName> resources = EntityMaterialsUtil.GetEntityMaterials(entity);
		foreach(ResourceName resource: resources)
		{
			AdjustMaterialTemperature(resource);
		}
	}

	private void UpdateParticles()
	{
		if (cameras.Count() > 0)
        {
            ParticleThermalModifier.AdjustParticleThermals();
        }
	}

	private void UpdateNearEntities()
	{
		GetGame().GetWorld().QueryEntitiesBySphere(ownerEnt.GetOrigin(), RADIUS, OnEntityFound, FilterEntities, EQueryEntitiesFlags.DYNAMIC);
	}

	private bool OnEntityFound(IEntity ent)
	{
		AdjustEntityProperties(ent, cameras.Count() > 0, true);
		return true;
	}

	private void AdjustEntityProperties(IEntity entity, bool adjustThermals, bool recursively)
	{
		entity.SetVComponentFlags(VCFlags.NOFILTER);
		entity.SetFlags(EntityFlags.SET_RENDERED);
		entity.Update();

		// iterate through children
		if (recursively)
		{
			IEntity child = entity.GetChildren();
			while (child)
			{
				AdjustEntityProperties(child, adjustThermals, true);
				child = child.GetSibling();
			}
		}
		
		if (Vehicle.Cast(entity))
		{
			array<CameraBase> cameraList = {};
			GetGame().GetCameraManager().GetCamerasList(cameraList);
			foreach(CameraBase camera: cameraList)
			{
				if (camera.IsSphereVisible(entity.GetOrigin(), 20))
				{
					entity.SetFixedLOD(0);
					return;
				}
			}
			entity.SetFixedLOD(-1);
		}
	}

	private void AdjustMaterialTemperature(ResourceName resourceName)
	{
		/*if (modifiedResources.Contains(resourceName))
		{
			return;
		}*/
		Material mat = Material.GetOrLoadMaterial(resourceName, 0);
		if (!mat)
		{
			return;
		}
		mat.SetParam("TITemperature3",  Math.RandomInt(190,200));
		mat.SetParam("TITemperatureCoef3",  Math.RandomFloat(0.9,1));
		modifiedResources.Insert(resourceName);
	}

	private bool FilterEntities(IEntity ent)
	{
		if (ent.GetID() == ownerEnt.GetID())
			return false;

		if (IsDestroyed(ent))
			return false;

		if (ent.FindComponent(VehicleBaseSimulation))
			return true;

		if (SCR_ChimeraCharacter.Cast(ent) != null)
			return true;

		return false;
	}

	private bool IsDestroyed(IEntity entity)
	{
		DamageManagerComponent damageManager = DamageManagerComponent.Cast(entity.FindComponent(DamageManagerComponent));
		return damageManager && damageManager.IsDestroyed();
	}
}

modded class Vehicle
{
	void Vehicle(IEntitySource src, IEntity parent)
	{
		GetGame().GetCallqueue().CallLater(OnVehicleCreated, 1000)
	}
	
	void OnVehicleCreated()
	{
		RVX_FLIR_Util.getInstance().UpdateMaterialsThermals();
	}
}

modded class SCR_ChimeraCharacter
{
	void SCR_ChimeraCharacter(IEntitySource src, IEntity parent)
	{
		GetGame().GetCallqueue().CallLater(OnCharacterCreated, 1000)
	}
	
	void OnCharacterCreated()
	{
		RVX_FLIR_Util.getInstance().UpdateMaterialsThermalsForCharacter(this);
	}
}
