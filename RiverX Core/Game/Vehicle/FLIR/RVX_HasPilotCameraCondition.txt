//------------------------------------------------------------------------------------------------
//! Returns true if current vehicle engine is on
[BaseContainerProps()]
class RVX_HasPilotCameraCondition : SCR_AvailableActionCondition
{
	//------------------------------------------------------------------------------------------------
	override bool IsAvailable(notnull SCR_AvailableActionsConditionData data)
	{
		if (!data)
			return false;

		IEntity entity = data.GetCurrentVehicle();
		if (!entity)
			return false;


		RVX_HasPilotCameraComponent hasPilotCamera = RVX_HasPilotCameraComponent.Cast(entity.FindComponent(RVX_HasPilotCameraComponent));
		return hasPilotCamera != null;
	}
};
