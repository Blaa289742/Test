[BaseContainerProps(), SCR_BaseManualCameraComponentTitle()]
class RVX_OrbitingManualCameraComponent: SCR_OrbitingManualCameraComponent
{
	private IEntity target;
	
	void SetTargetEntity(IEntity targetEnt)
	{
		target = targetEnt;
	}
	
	override bool GetCenterPos()
	{
		if(target)
		{
			super.SetTargetVector(target.GetOrigin());
			return true;
		} else
		{
			return super.GetCenterPos();
		}
	}
}

[BaseContainerProps(), SCR_BaseManualCameraComponentTitle()]
modded class SCR_OrbitingManualCameraComponent
{
	void SetTargetVector(vector target)
	{
		m_vTarget = target;
	}
}
