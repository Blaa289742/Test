class RVX_SlotAnimMapperClass : ScriptGameComponentClass
{

}

class RVX_SlotAnimMapper : ScriptGameComponent
{
	[Attribute(desc: "Holds an array of slot mapping groups.")]
	protected ref array<ref RVX_SlotAnimMapping> slotMappings;
	
	protected SignalsManagerComponent parentSignalManager;
	protected SlotManagerComponent slotManager;
	
	override void OnPostInit(IEntity owner)
    {
        super.OnPostInit(owner);
		parentSignalManager = SignalsManagerComponent.Cast(owner.FindComponent(SignalsManagerComponent));
		slotManager = SlotManagerComponent.Cast(owner.FindComponent(SlotManagerComponent));
		
		parentSignalManager.AddOrFindSignal("handBrake");
		parentSignalManager.AddOrFindSignal("engineRPM");
		parentSignalManager.AddOrFindSignal("engineOn");
		
		if(parentSignalManager && slotManager)
			SetEventMask(owner, EntityEvent.SIMULATE);
    }
	
	override protected void EOnSimulate(IEntity owner, float timeSlice)
    {
		foreach (RVX_SlotAnimMapping slotMapping: slotMappings)
		{
			EntitySlotInfo slotInfo = slotManager.GetSlotByName(slotMapping.slotName);
			if(slotInfo)
			{
				IEntity slotEntity = slotInfo.GetAttachedEntity();
				if (!slotEntity)
					continue;
				SignalsManagerComponent childSignalManager = SignalsManagerComponent.Cast(slotEntity.FindComponent(SignalsManagerComponent));
				if(childSignalManager)
				{
					foreach(RVX_AnimSignalMapping signalMapping: slotMapping.signalMappings)
					{
						int parentSignalIndex = parentSignalManager.FindSignal(signalMapping.parentSignal);
						float parentSignalValue = parentSignalManager.GetSignalValue(parentSignalIndex);
						
						int childSignalIndex = childSignalManager.FindSignal(signalMapping.childSignal);
						float childSignalValue = childSignalManager.GetSignalValue(childSignalIndex);
						if (parentSignalValue != childSignalValue)
						{
							childSignalManager.SetSignalValue(childSignalIndex, parentSignalValue);
						}					
					}
				}
			}
		}
    }
}

[BaseContainerProps()]
class RVX_SlotAnimMapping : Managed
{
	[Attribute("1")]
	string slotName;
	
	[Attribute("2")]
	ref array<ref RVX_AnimSignalMapping> signalMappings;
}

[BaseContainerProps()]
class RVX_AnimSignalMapping : Managed
{
	[Attribute("")]
	string parentSignal;
	
	[Attribute("")]
	string childSignal;
}
