class RVX_ChildToParentSignalProviderClass : ScriptGameComponentClass
{

}

class RVX_ChildToParentSignalProvider : ScriptGameComponent
{
	[Attribute(desc: "Holds an array of slot signals from child")]
	protected ref array<string> signals;
	
	protected SignalsManagerComponent parentSignalManager;
	protected SignalsManagerComponent childSignalManager;
	
	override void OnPostInit(IEntity owner)
    {
        super.OnPostInit(owner);
		IEntity parent = GetOwner().GetParent();
		childSignalManager = SignalsManagerComponent.Cast(GetOwner().FindComponent(SignalsManagerComponent));
		if (parent)
		{
			parentSignalManager = SignalsManagerComponent.Cast(GetOwner().GetParent().FindComponent(SignalsManagerComponent));
			
			foreach(string signal: signals)
			{
				parentSignalManager.AddOrFindSignal(signal);			
				childSignalManager.AddOrFindSignal(signal);			
			}
			
			SetEventMask(owner, EntityEvent.FRAME);
		}			
    }
	
	override protected void EOnFrame(IEntity owner, float timeSlice)
    {
		foreach(string signal: signals)
		{
			int parentSignalIndex = parentSignalManager.FindSignal(signal);
			float parentSignalValue = parentSignalManager.GetSignalValue(parentSignalIndex);
			
			int childSignalIndex = childSignalManager.FindSignal(signal);
			float childSignalValue = childSignalManager.GetSignalValue(childSignalIndex);
			
			if (parentSignalValue != childSignalValue)
			{
				parentSignalManager.SetSignalValue(parentSignalIndex, childSignalValue);
			}		
		}
    }
}
