class RVX_LaserDesignatorComponentClass: ScriptComponentClass
{

}
class RVX_LaserDesignatorComponent : ScriptComponent
{
	[Attribute("LaserDesignatorToggle", UIWidgets.Auto, "Rocket", "et")]
	string laserToggleAction;
	
	[Attribute("false", UIWidgets.Auto, "Rocket", "et")]
	bool isToggleAction;
	
	[Attribute("false", UIWidgets.Auto, "Rocket", "et")]
	bool enableDebugLaserTarget;
	
	[Attribute("4000", UIWidgets.Auto, "Rocket", "et")]
	float maxLaserDistance;
	
	protected IEntity instance;
	
	protected ref array<ref Shape> m_aDebugShapes = {};
	
	protected float laserDistance = 0;
	protected string laserCode = "1688";
	protected IEntity laserEmitterEnt; 
	private bool laserActive = false;
	private ResourceName laserOnHUDLayout = "{8899232525A6FC6D}UI/layouts/HUD/LaserOnOverlay.layout";
	private FrameWidget laserONWidget;
	
	private IEntity cameraEntity;
	
	float GetLaserDistance()
	{
		return Math.Round(laserDistance * maxLaserDistance);
	}
	
	string GetLaserCode()
	{
		return laserCode;
	}
	
	void SetLaserCode(string newCode)
	{
	 	laserCode = newCode;
		UpdateLaserCode();
	}
	
	void OnLaserCodeSynced(string newCode)
	{
	 	laserCode = newCode;
	}
	
	void RegisterActions()
	{
		SetEventMask(laserEmitterEnt, EntityEvent.FRAME);
		GetGame().GetInputManager().AddActionListener(laserToggleAction, EActionTrigger.DOWN, OnToggleLaserDesignator);
		if (!isToggleAction)
		{
			GetGame().GetInputManager().AddActionListener(laserToggleAction, EActionTrigger.UP, OnToggleLaserDesignator);
		}
	}
	
	void UnRegisterActions()
	{
		ClearEventMask(laserEmitterEnt, EntityEvent.FRAME);
		GetGame().GetInputManager().RemoveActionListener(laserToggleAction, EActionTrigger.DOWN, OnToggleLaserDesignator);
		if (!isToggleAction)
		{
			GetGame().GetInputManager().AddActionListener(laserToggleAction, EActionTrigger.UP, OnToggleLaserDesignator);
		}
		DeleteLaserTarget();
	}
	
	void SetCameraEntity(IEntity entity)
	{
		cameraEntity = entity;
	}
	
	bool IsLaserActive()
	{
		return laserActive;
	}
	
	//------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
	//------------------------------------------------------------------------------------------------
	override protected void OnPostInit(IEntity owner)
	{
		super.OnPostInit(owner);
		laserEmitterEnt = owner;
		SetEventMask(owner, EntityEvent.INIT);
		PreloadWidgets();
	}

	//------------------------------------------------------------------------------------------------
	override protected void EOnInit(IEntity owner)
	{
        super.OnPostInit(owner);
    }
	
	override protected void EOnFrame(IEntity owner, float timeSlice)
	{
		UpdateLaserTarget();
		laserONWidget.SetVisible(laserActive);
	}
	
	private void PreloadWidgets()
	{
		Widget root = GetGame().GetWorkspace().CreateWidgets(laserOnHUDLayout);
		root.SetVisible(true);
		root.SetEnabled(true);
		
		laserONWidget = FrameWidget.Cast(root.FindAnyWidget("LaserON"));
		laserONWidget.SetVisible(false);
	}
	
	protected void OnToggleLaserDesignator(float value, EActionTrigger reason)
	{
		if(isToggleAction)
			laserActive = !laserActive;
		else 
			laserActive = reason == EActionTrigger.DOWN;
		
		
		if(laserActive)
		{
			CreateLaserTarget()
		} else 
		{
			DeleteLaserTarget();
		}
	}
	
	protected void UpdateLaserTarget()
	{
		if(!laserActive) 
			return;
			
		vector laserPos = GetCameraTargetPos();
		IEntity player = GetGame().GetPlayerController().GetControlledEntity();
		RplId creatorId = RplComponent.Cast(GetOwner().FindComponent(RplComponent)).Id();
       	SCR_ChimeraCharacter character = SCR_ChimeraCharacter.Cast(player);
		character.UpdateLaserTargetPosAndCode(laserPos, creatorId, laserCode);
		
		if (enableDebugLaserTarget) {
			m_aDebugShapes.Clear();	
			
			Shape targetSphere = Shape.CreateSphere(Color.RED, ShapeFlags.DEFAULT, laserPos, 3);
			m_aDebugShapes.Insert(targetSphere);
			
		}
	}
	
	protected void UpdateLaserCode()
	{
		RplId creatorId = RplComponent.Cast(GetOwner().FindComponent(RplComponent)).Id();
       	SCR_ChimeraCharacter character = GetSyncUpdater();
		character.UpdateLaserCode(creatorId, laserCode);
	}
	
	protected void DeleteLaserTarget()
	{
		IEntity player = GetGame().GetPlayerController().GetControlledEntity();
		RplId creatorId = RplComponent.Cast(GetOwner().FindComponent(RplComponent)).Id();
	    SCR_ChimeraCharacter character = SCR_ChimeraCharacter.Cast(player);
	    character.DeleteLaserTarget(creatorId);
		
		if (enableDebugLaserTarget)
			m_aDebugShapes.Clear();		
	}
	
	protected void CreateLaserTarget()
	{
		IEntity player = GetGame().GetPlayerController().GetControlledEntity();
		RplId creatorId = RplComponent.Cast(GetOwner().FindComponent(RplComponent)).Id();
        SCR_ChimeraCharacter character = SCR_ChimeraCharacter.Cast(player);
        character.SpawnLaserTarget(creatorId, laserCode);
	}
	
	protected vector GetCameraTargetPos()
	{
		IEntity camera;
		if (cameraEntity)
		{
			camera = cameraEntity;
		} else 
		{
			camera = GetGame().GetCameraManager().CurrentCamera();
		}
		auto direction = camera.GetYawPitchRoll().AnglesToVector();
		auto distance = RayCastFromCamera(direction, camera);
		laserDistance = distance;
		
		return camera.GetOrigin() + (direction * maxLaserDistance * distance);
	}
	
	
	private float RayCastFromCamera(vector direction, IEntity camera)
	{		
		autoptr TraceParam param = new TraceParam;
		param.LayerMask = EPhysicsLayerDefs.Projectile;
		param.Flags = TraceFlags.ENTS | TraceFlags.VISIBILITY | TraceFlags.WORLD | TraceFlags.OCEAN;
		param.Start = camera.GetOrigin() + (direction * 2);
		param.End = camera.GetOrigin() + (direction * maxLaserDistance);

		return GetGame().GetWorld().TraceMove(param, null);
	}
	
	private SCR_ChimeraCharacter GetSyncUpdater()
	{
		IEntity player = GetGame().GetPlayerController().GetControlledEntity();
       	SCR_ChimeraCharacter character = SCR_ChimeraCharacter.Cast(player);
		return character;
	}
}

modded class SCR_ChimeraCharacter
{ 
	void SpawnLaserTarget(RplId creatorId, string laserCode)
    { 
        Rpc(RpcAsk_SpawnLaserTarget, creatorId, laserCode);
    }
	
    [RplRpc(RplChannel.Reliable, RplRcver.Server)]
    private void RpcAsk_SpawnLaserTarget(RplId creatorId, string laserCode)
    {	
		RpcAsk_BroadcastSpawnLaserTarget(creatorId, laserCode);
		Rpc(RpcAsk_BroadcastSpawnLaserTarget, creatorId, laserCode);
    }
	
	[RplRpc(RplChannel.Reliable, RplRcver.Broadcast)]
    private void RpcAsk_BroadcastSpawnLaserTarget(RplId creatorId, string laserCode)
    {	
		RVX_LaserTargetManager.getInstance().CreateLaserTarget(creatorId, laserCode);
    }
	///////////////////////////////////////////////////////////////////////////
	
	void DeleteLaserTarget(RplId creatorId)
	{
		Rpc(RpcAsk_DeleteLaserTarget, creatorId);
	}
	
	[RplRpc(RplChannel.Reliable, RplRcver.Server)]
    private void RpcAsk_DeleteLaserTarget(RplId creatorId)
    {
		RpcAsk_BroadcastDeleteLaserTarget(creatorId);
		Rpc(RpcAsk_BroadcastDeleteLaserTarget, creatorId);
	}
	
	[RplRpc(RplChannel.Reliable, RplRcver.Broadcast)]
    private void RpcAsk_BroadcastDeleteLaserTarget(RplId creatorId)
    {	
		RVX_LaserTargetManager.getInstance().RemoveLaserTarget(creatorId);
    }
	///////////////////////////////////////////////////////////////////////////
	
	void UpdateLaserTargetPosAndCode(vector position, RplId creatorId, string laserCode)
    { 
        Rpc(RpcAsk_UpdateLaserTargetPosAndCode, position, creatorId, laserCode);
    }
		
    [RplRpc(RplChannel.Reliable, RplRcver.Server)]
    private void RpcAsk_UpdateLaserTargetPosAndCode(vector position, RplId creatorId, string laserCode)
    {
		RpcAsk_BroadcastLaserPositionAndCode(position, creatorId, laserCode);
		Rpc(RpcAsk_BroadcastLaserPositionAndCode, position, creatorId, laserCode);
    }
	
	[RplRpc(RplChannel.Reliable, RplRcver.Broadcast)]
    private void RpcAsk_BroadcastLaserPositionAndCode(vector position, RplId creatorId, string laserCode)
    {
       	RVX_LaserTargetManager.getInstance().UpdateLaserTarget(creatorId, position, laserCode);
    }
	//////////////////////////////////////////////////////////////////////////////
	void UpdateLaserCode(RplId creatorId, string laserCode)
    { 
        Rpc(RpcAsk_UpdateLaserCode, creatorId, laserCode);
    }
		
    [RplRpc(RplChannel.Reliable, RplRcver.Server)]
    private void RpcAsk_UpdateLaserCode(RplId creatorId, string laserCode)
    {
		RpcAsk_BroadcastUpdateLaserCode(creatorId, laserCode);
		Rpc(RpcAsk_BroadcastUpdateLaserCode, creatorId, laserCode);
    }
	
	[RplRpc(RplChannel.Reliable, RplRcver.Broadcast)]
    private void RpcAsk_BroadcastUpdateLaserCode(RplId creatorId, string laserCode)
    {
       	RplComponent comp = RplComponent.Cast(Replication.FindItem(creatorId));
		if(comp)
		{
			IEntity vehicle = comp.GetEntity();
			RVX_LaserDesignatorComponent laserDesignator = Inject<RVX_LaserDesignatorComponent>.Get(vehicle);
			laserDesignator.OnLaserCodeSynced(laserCode);
		}
    }
}
